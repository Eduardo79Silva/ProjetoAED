119 unsigned glyphLength(const std::string& s) const {
120 unsigned int _byteLength = s.length();
121#ifdef TEXTTABLE_ENCODE_MULTIBYTE_STRINGS
122 #ifdef TEXTTABLE_USE_EN_US_UTF8
123 std::setlocale(LC_ALL, "en_US.utf8");
124#else
125#error You need to specify the encoding if the TextTable library uses multybyte string encoding!
126#endif
127 unsigned int u = 0;
128 const char *c_str = s.c_str();
129 unsigned _glyphLength = 0;
130 while (u < _byteLength) {
131 u += std::mblen(&c_str[u], _byteLength - u);
132 _glyphLength += 1;
133 }
134 return _glyphLength;
135#else
136 return _byteLength;
137#endif

140 void determineWidths() const {
141 if (_rows.empty()) {
142 return;
143 }
144 _width.assign(columns(), 0);
145 _utf8width.assign(columns(), 0);
146 for (auto rowIterator = _rows.begin(); rowIterator != _rows.end();
147 ++rowIterator) {
148 Row const &row = *rowIterator;
149 for (unsigned i = 0; i < row.size(); ++i) {
150 _width[i] =
151 _width[i] > glyphLength(row[i]) ? _width[i] : glyphLength(row[i]);
152 }
153 }
154 }

161 if (_alignment.find(i) == _alignment.end()) {
162 _alignment[i] = Alignment::LEFT;
163 }

168inline std::ostream &operator<<(std::ostream &stream, const TextTable& table) {
169 if (table.rows().empty()) {
170 return stream;
171 }
172 table.setup();
173 if (table.has_ruler()) {
174 stream << table.ruler() << "\n";
175 }
176 for (auto rowIterator = table.rows().begin();
177 rowIterator != table.rows().end(); ++rowIterator) {
178 TextTable::Row const &row = *rowIterator;
179 stream << table.vertical();
180 for (unsigned i = 0; i < row.size(); ++i) {
181 auto alignment = table.alignment(i) == TextTable::Alignment::LEFT
182 ? std::left
183 : std::right;
184 // std::setw( width ) works as follows: a string which goes in the stream
185 // with byte length (!) l is filled with n spaces so that l+n=width. For a
186 // utf8 encoded string the glyph length g might be smaller than l. We need
187 // n spaces so that g+n=width which is equivalent to g+n+l-l=width ==> l+n
188 // = width+l-g l-g (that means glyph length minus byte length) has to be
189 // added to the width argument. l-g is computed by correctDistance.
190 stream << std::setw(table.width(i) + table.correctDistance(row[i]))
191 << alignment << row[i];
192 stream << table.vertical();
193 }
194 stream << "\n";
195 if (table.has_ruler()) {
196 stream << table.ruler() << "\n";
197 }
198 }
199
200 return stream;
201}
